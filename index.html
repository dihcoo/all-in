// ── Updated Audio Core Logic ── let audioCtx, analyser, sourceNode; let fbqStage, afs2Parallel, bbeHi, bbeLo, bbeDelay, bbeGain, master, afs2Gain; let isProcess = true, playlist = [], currentIdx = -1; const audioEl = document.getElementById('audio-el'); async function initAudio() { // Only initialize the context and nodes ONCE if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); analyser = audioCtx.createAnalyser(); analyser.fftSize = 256; // Ensure the audio element is ready for Web Audio sourceNode = audioCtx.createMediaElementSource(audioEl); // 1. FBQ1000 Stage (Series) fbqStage = audioCtx.createBiquadFilter(); fbqStage.type = "notch"; fbqStage.frequency.value = 800; fbqStage.Q.value = 10; // 2. DBX AFS2 Stage (Parallel Path) afs2Parallel = audioCtx.createBiquadFilter(); afs2Parallel.type = "notch"; afs2Parallel.frequency.value = 1200; afs2Gain = audioCtx.createGain(); afs2Gain.gain.value = 0.4; // 3. BBE 882 Emulation Stage bbeHi = audioCtx.createBiquadFilter(); bbeHi.type = "highshelf"; bbeHi.frequency.value = 3500; bbeLo = audioCtx.createBiquadFilter(); bbeLo.type = "lowshelf"; bbeLo.frequency.value = 150; bbeDelay = audioCtx.createDelay(0.1); bbeGain = audioCtx.createGain(); master = audioCtx.createGain(); // ── SIGNAL ROUTING ── // Source -> FBQ1000 (Series) sourceNode.connect(fbqStage); // Path A: BBE Sonic Maximizer Path fbqStage.connect(bbeHi); bbeHi.connect(bbeLo); bbeLo.connect(bbeDelay); bbeDelay.connect(bbeGain); // Path B: DBX AFS2 (Parallel) fbqStage.connect(afs2Parallel); afs2Parallel.connect(afs2Gain); // Sum Paths -> Master -> Analyser -> Output bbeGain.connect(master); afs2Gain.connect(master); master.connect(analyser); analyser.connect(audioCtx.destination); } // Always attempt to resume the context on a user gesture if (audioCtx.state === 'suspended') { await audioCtx.resume(); } } // ── Optimized Playback Function ── window.playTrack = async (i) => { try { await initAudio(); // Ensure engine is alive currentIdx = i; // Only update source if it's a new track if (audioEl.src !== playlist[i].url) { audioEl.src = playlist[i].url; audioEl.load(); } const playPromise = audioEl.play(); if (playPromise !== undefined) { playPromise.then(() => { document.getElementById('now-playing').textContent = playlist[i].name; document.getElementById('status-text').textContent = "PROCESSING..."; updateParams(); updateList(); }).catch(error => { console.error("Playback failed:", error); document.getElementById('status-text').textContent = "PLAYBACK BLOCKED"; }); } } catch (e) { console.error("Audio init error:", e); } };
